# ask_user.form

title {
  Modify a user account
}

field {
  id       = UNAME
  label    = Login name
  type     = STRING
  required = YES
  len      = 8
  hscroll  = YES
  list_cmd = command:single-val:cut -d : -f 1,5 --output-delimiter=" " /etc/passwd | sort
}

action { form:usermod %{UNAME} }
